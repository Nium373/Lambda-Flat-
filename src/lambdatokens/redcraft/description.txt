LexerRule:
  // Test the text and advance text if matched
  abstract Optional<Token> apply(Text remaining)
  int length
  int precedence
Lexer:
  List<LexerRule> rules
  tokenize(Text text):
    sort rules by LexerRule#length (longest first), then precedence
    List output;

    while text has remaining:
      for rule in rules:
        match = rule.apply(text)
        if(match is present):
          output.add(match)
          break

    return output

StructuredToken:
  parent, child, previous, next

  toTree(Set<Type> down, Set<Type> up):
    list = this;
    while(list != null):
      if(down contains list.type):
        list.child = list.next
        list.next = null
        list.child.toTree(down, up)
        break
      elif(up contains list.type):
        list.parent.next = list.next;
        list.next = null
      else
        list = list.next
